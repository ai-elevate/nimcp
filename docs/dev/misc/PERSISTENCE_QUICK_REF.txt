================================================================================
USAGE EXAMPLES
================================================================================

// Save brain state
brain_save(brain, "model_v1.nimcp");

// Load brain state
brain_t loaded = brain_load("model_v1.nimcp");

// Create snapshot
brain_save_snapshot(brain, "experiment_1", "Before hyperparameter tuning");

// Restore snapshot
brain_t restored = brain_restore_snapshot(NULL, "experiment_1");

// List snapshots
brain_snapshot_info_t infos[100];
uint32_t count;
brain_list_snapshots(brain, infos, 100, &count);
for (uint32_t i = 0; i < count; i++) {
    printf("%s: %s (size: %u bytes)\n", 
           infos[i].name, infos[i].description, infos[i].file_size);
}

// Delete snapshot
brain_delete_snapshot(brain, "experiment_1");

================================================================================
SNAPSHOT FORMAT
================================================================================

Snapshots are saved to: {snapshot_dir}/{name}_{timestamp}.snapshot
Metadata is saved to:   {snapshot_dir}/{name}_{timestamp}.snapshot.info

Default snapshot_dir: "./snapshots" (or brain->config.snapshot_dir)

Metadata file format (text):
name=experiment_v1
timestamp=1642531200
description=Before parameter tuning
compressed=1
encrypted=0
