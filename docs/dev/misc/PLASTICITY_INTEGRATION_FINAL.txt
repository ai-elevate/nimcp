EXAMPLE INTEGRATION
================================================================================

From nimcp_neuromodulators.c:

// Module Registration (in neuromodulator_system_create)
system->bio_ctx = NULL;
system->bio_async_enabled = false;
bio_async_context_t* ctx = bio_router_get_global_context();
if (ctx) {
    system->bio_ctx = ctx;
    system->bio_async_enabled = bio_router_register_module(
        ctx, BIO_MODULE_NEUROMODULATOR, "neuromodulator_system_create");
}

// Event Broadcasting (on neuromodulator release)
if (system && system->bio_async_enabled && system->bio_ctx) {
    bio_router_broadcast(system->bio_ctx, BIO_MODULE_NEUROMODULATOR,
                        BIO_MSG_PLASTICITY_UPDATE, NULL, 0);
}

// Module Unregistration (in cleanup)
if (system && system->bio_async_enabled && system->bio_ctx) {
    bio_router_unregister_module(system->bio_ctx, BIO_MODULE_PLASTICITY);
}
